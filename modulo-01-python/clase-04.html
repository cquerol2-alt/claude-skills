📖 Resumen

Funciones en Python

Las funciones son bloques reutilizables de código que realizan una tarea específica. Se definen con def y pueden recibir parámetros y devolver valores.

Definición y parámetros

def saludar(nombre, apellido):
    return f"Hola, {nombre} {apellido}"

resultado = saludar("Juan", "Querol")
print(resultado)  # Hola, Juan Querol

Valores por defecto

def crear_usuario(nombre, rol="usuario", activo=True):
    return {"nombre": nombre, "rol": rol, "activo": activo}

admin = crear_usuario("Carlos", rol="admin")
print(admin)  # {'nombre': 'Carlos', 'rol': 'admin', 'activo': True}

*args y **kwargs

# *args: argumentos posicionales variables
def sumar(*numeros):
    return sum(numeros)

print(sumar(1, 2, 3, 4, 5))  # 15

# **kwargs: argumentos nombrados variables
def crear_perfil(nombre, **datos):
    return {**datos, "nombre": nombre}

perfil = crear_perfil("Ana", edad=28, ciudad="Madrid", rol="dev")
print(perfil)  # {'edad': 28, 'ciudad': 'Madrid', 'rol': 'dev', 'nombre': 'Ana'}

Funciones lambda

# Lambda: función anónima en una línea
duplicar = lambda x: x * 2
print(duplicar(5))  # 10

# Útiles con map, filter, sorted
numeros = [1, 2, 3, 4, 5]
pares = list(filter(lambda x: x % 2 == 0, numeros))
print(pares)  # [2, 4]

Módulos e imports

Los módulos permiten organizar el código en archivos reutilizables. Python tiene módulos built-in y puedes crear los tuyos.

Importar módulos built-in

import math
from datetime import datetime
from random import randint, choice

print(math.pi)  # 3.141592653589793
ahora = datetime.now()
numero = randint(1, 100)

Crear tu propio módulo

utils.py

def saludar(nombre):
    return f"¡Hola, {nombre}!"

def despedir(nombre):
    return f"¡Hasta pronto, {nombre}!"

PI = 3.14159

main.py

import utils
from utils import saludar, PI

print(utils.saludar("María"))  # ¡Hola, María!
print(saludar("Juan"))  # ¡Hola, Juan!
print(PI)  # 3.14159

__name__ == "__main__"

# archivo: herramientas.py
def procesar_datos(datos):
    return [x * 2 for x in datos]

# Este código solo se ejecuta cuando corres el archivo directamente
if __name__ == "__main__":
    datos = [1, 2, 3, 4, 5]
    resultado = procesar_datos(datos)
    print(resultado)  # [2, 4, 6, 8, 10]

Comparación con JavaScript

Concepto Python JavaScript
Función básica def func(x): return x*2 function func(x) { return x*2; }
Parámetros por defecto def func(x=5): function func(x=5) {}
Función anónima lambda x: x*2 (x) => x*2
Múltiples argumentos def func(*args): function func(...args) {}
Importar módulo import modulo import { func } from 'modulo'

💻 Práctica Guiada

Paso 1: Crear funciones de utilidad

Crea un archivo texto.py con funciones para manipular texto:

# texto.py
def contar_palabras(texto):
    """Cuenta el número de palabras en un texto"""
    return len(texto.split())

def capitalizar_titulo(texto):
    """Convierte texto a formato título"""
    return texto.title()

def invertir_texto(texto):
    """Invierte un texto"""
    return texto[::-1]

def contar_caracteres(texto, incluir_espacios=False):
    """Cuenta caracteres, opcionalmente excluyendo espacios"""
    if incluir_espacios:
        return len(texto)
    return len(texto.replace(" ", ""))

Paso 2: Crear un módulo reutilizable

Ahora crea operaciones.py con funciones matemáticas:

# operaciones.py
def sumar(a, b):
    return a + b

def restar(a, b):
    return a - b

def multiplicar(a, b):
    return a * b

def dividir(a, b):
    if b == 0:
        return "Error: no se puede dividir entre cero"
    return a / b

def potencia(base, exponente=2):
    return base ** exponente

Paso 3: Usar los módulos

Crea main.py que importe y use ambos módulos:

# main.py
import texto
import operaciones

# Usar módulo texto
parrafo = "Python es un lenguaje de programación"
print(f"Palabras: {texto.contar_palabras(parrafo)}")
print(f"Título: {texto.capitalizar_titulo(parrafo)}")
print(f"Invertido: {texto.invertir_texto(parrafo)}")
print(f"Caracteres (sin espacios): {texto.contar_caracteres(parrafo)}")

print("\n--- Operaciones Matemáticas ---\n")

# Usar módulo operaciones
x, y = 10, 3
print(f"{x} + {y} = {operaciones.sumar(x, y)}")
print(f"{x} - {y} = {operaciones.restar(x, y)}")
print(f"{x} * {y} = {operaciones.multiplicar(x, y)}")
print(f"{x} / {y} = {operaciones.dividir(x, y)}")
print(f"{x} ^ {y} = {operaciones.potencia(x, y)}")

Paso 4: Mejorar con __name__ == "__main__"

Modifica operaciones.py para que funcione como módulo Y como script:

# operaciones.py (mejorado)
def sumar(a, b):
    return a + b

def restar(a, b):
    return a - b

def multiplicar(a, b):
    return a * b

def dividir(a, b):
    if b == 0:
        return "Error: no se puede dividir entre cero"
    return a / b

if __name__ == "__main__":
    # Este código solo se ejecuta al correr operaciones.py directamente
    print("Probando el módulo operaciones:")
    print(f"5 + 3 = {sumar(5, 3)}")
    print(f"5 - 3 = {restar(5, 3)}")
    print(f"5 * 3 = {multiplicar(5, 3)}")
    print(f"5 / 3 = {dividir(5, 3)}")

🧪 Mini-reto

Crea un módulo calculadora

Tu tarea: crear un módulo de calculadora completo con su propio archivo de pruebas.

Requisitos

  • Archivo calculadora.py con funciones: add, subtract, multiply, divide
  • Cada función debe validar los argumentos (ej: no dividir entre 0)
  • Archivo test_calculadora.py que importe y pruebe las funciones
  • Archivo main.py que use la calculadora de forma interactiva

Estructura esperada

proyecto/
├── calculadora.py
├── test_calculadora.py
└── main.py

Ejemplo de calculadora.py

# calculadora.py
def add(a, b):
    """Suma dos números"""
    return a + b

def subtract(a, b):
    """Resta dos números"""
    return a - b

def multiply(a, b):
    """Multiplica dos números"""
    return a * b

def divide(a, b):
    """Divide dos números"""
    if b == 0:
        raise ValueError("No se puede dividir entre cero")
    return a / b

if __name__ == "__main__":
    print("Calculadora cargada correctamente")

Ejemplo de test_calculadora.py

# test_calculadora.py
from calculadora import add, subtract, multiply, divide

print("=== PRUEBAS DE CALCULADORA ===\n")

# Suma
assert add(5, 3) == 8
print("✓ add(5, 3) = 8")

# Resta
assert subtract(10, 4) == 6
print("✓ subtract(10, 4) = 6")

# Multiplicación
assert multiply(6, 7) == 42
print("✓ multiply(6, 7) = 42")

# División
assert divide(20, 4) == 5
print("✓ divide(20, 4) = 5")

# División por cero
try:
    divide(10, 0)
except ValueError:
    print("✓ divide(10, 0) lanza ValueError")

print("\n¡Todas las pruebas pasaron!")

Ejemplo de main.py

# main.py
from calculadora import add, subtract, multiply, divide

print("=== CALCULADORA INTERACTIVA ===\n")

while True:
    operacion = input("Operación (add/sub/mul/div) o 'salir': ").lower()

    if operacion == "salir":
        print("¡Hasta luego!")
        break

    try:
        a = float(input("Primer número: "))
        b = float(input("Segundo número: "))

        if operacion == "add":
            print(f"Resultado: {add(a, b)}\n")
        elif operacion == "sub":
            print(f"Resultado: {subtract(a, b)}\n")
        elif operacion == "mul":
            print(f"Resultado: {multiply(a, b)}\n")
        elif operacion == "div":
            print(f"Resultado: {divide(a, b)}\n")
        else:
            print("Operación no reconocida\n")
    except ValueError:
        print("Error: ingresa números válidos\n")
